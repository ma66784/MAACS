## Function for supporting premise statement 1
premise_1 <- function(dataset) {
  q <- stats::quantile(dataset$age, c(0.25, 0.75), na.rm = TRUE)
  young <- dataset$fev1[dataset$age <= q[1]]
  old <- dataset$fev1[dataset$age >= q[2]]
  if (mean(old, na.rm = TRUE) > mean(young, na.rm = TRUE)) return(TRUE)
  else return(FALSE) 
}

## Function for supporting premise statement 3
premise_3 <- function(dataset, alpha = 0.05) {
  med_age <- stats::median(dataset$age, na.rm = TRUE)
  young <- dataset$fev1[dataset$age <= med_age]
  old   <- dataset$fev1[dataset$age >  med_age]
  
  tt <- stats::t.test(old, young, alternative = "greater", var.equal = FALSE)
  
  (mean(old, na.rm = TRUE) > mean(young, na.rm = TRUE)) && 
    (!is.na(tt$p.value) && tt$p.value < alpha)
}
